> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Day Plan feature — an actionable daily plan (hybrid time-blocks + backlog) generated after briefing, calendar-aware with conflict detection, editable via Desktop UI, regenerable with user feedback that preserves manually-added items.

**Architecture:** New Go package `internal/dayplan/` (pipeline, gather, merge, conflicts) + new DB tables (`day_plans`, `day_plan_items`, migration v65) + new prompt template (`day_plan.generate`) + new CLI `watchtower day-plan {show|list|generate|reset|check-conflicts}` + daemon phases 7 (generation) and 8 (calendar items sync + conflict detection). Swift Desktop adds a dedicated tab with timeline + backlog UI that mutates DB directly via GRDB and shells out to CLI for regeneration.

**Tech Stack:** Go 1.25 (modernc.org/sqlite, cobra, viper), AI via `claude` / `codex` CLI subprocess using existing `digest.Generator` interface, testify + require for tests, SwiftUI + GRDB.swift + Yams for Desktop (macOS 14+).

    ID                int64

    PlanDate          string // YYYY-MM-DD, local date
    Status            string // active | archived
    HasConflicts      bool
    ConflictSummary   sql.NullString
    GeneratedAt       time.Time

    RegenerateCount   int
    FeedbackHistory   string // JSON []string
    PromptVersion     sql.NullString
    BriefingID        sql.NullInt64
    ReadAt            sql.NullTime
    CreatedAt         time.Time
    UpdatedAt         time.Time
}

// DayPlanItem is an individual item inside a day plan.
type DayPlanItem struct {
    ID          int64
    DayPlanID   int64
    Kind        string // timeblock | backlog
    SourceType  string // task | briefing_attention | jira | calendar | manual | focus
    SourceID    sql.NullString
    Title       string
    Description sql.NullString
    Rationale   sql.NullString
    StartTime   sql.NullTime
    EndTime     sql.NullTime
    DurationMin sql.NullInt64
    Priority    sql.NullString
    Status      string // pending | done | skipped
    OrderIndex  int
    Tags        string // JSON, may be empty string
    CreatedAt   time.Time
    UpdatedAt   time.Time
}

package db

import (
    "database/sql"
    "encoding/json"
    "fmt"
    "strings"
    "time"
)

// CreateDayPlan inserts a new day plan, returning its id.
func (db *DB) CreateDayPlan(p *DayPlan) (int64, error) {
    if p.FeedbackHistory == "" { p.FeedbackHistory = "[]" }
    res, err := db.Exec(`INSERT INTO day_plans
        (user_id, plan_date, status, has_conflicts, conflict_summary, generated_at,
         last_regenerated_at, regenerate_count, feedback_history, prompt_version, briefing_id, read_at)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
        p.UserID, p.PlanDate, p.Status, boolToInt(p.HasConflicts), p.ConflictSummary,
        p.GeneratedAt.UTC().Format(time.RFC3339), p.LastRegeneratedAt, p.RegenerateCount,
        p.FeedbackHistory, p.PromptVersion, p.BriefingID, p.ReadAt)
    if err != nil { return 0, fmt.Errorf("create day_plan: %w", err) }
    return res.LastInsertId()
}

// UpsertDayPlan inserts or updates a day plan by (user_id, plan_date).
func (db *DB) UpsertDayPlan(p *DayPlan) (int64, error) {
    existing, err := db.GetDayPlan(p.UserID, p.PlanDate)
    if err != nil { return 0, err }
    if existing != nil {
        p.ID = existing.ID
        _, err := db.Exec(`UPDATE day_plans SET
            status=?, has_conflicts=?, conflict_summary=?, generated_at=?,
            last_regenerated_at=?, regenerate_count=?, feedback_history=?,
            prompt_version=?, briefing_id=?, read_at=?,
            updated_at=strftime('%Y-%m-%dT%H:%M:%SZ','now')
            WHERE id=?`,
            p.Status, boolToInt(p.HasConflicts), p.ConflictSummary,
            p.GeneratedAt.UTC().Format(time.RFC3339), p.LastRegeneratedAt, p.RegenerateCount,
            p.FeedbackHistory, p.PromptVersion, p.BriefingID, p.ReadAt, existing.ID)
        if err != nil { return 0, fmt.Errorf("update day_plan: %w", err) }
        return existing.ID, nil
    }
    return db.CreateDayPlan(p)
}

func (db *DB) GetDayPlan(userID, date string) (*DayPlan, error) {
    row := db.QueryRow(`SELECT id, user_id, plan_date, status, has_conflicts, conflict_summary,
        generated_at, last_regenerated_at, regenerate_count, feedback_history, prompt_version,
        briefing_id, read_at, created_at, updated_at
        FROM day_plans WHERE user_id=? AND plan_date=?`, userID, date)
    return scanDayPlan(row)
}

func (db *DB) GetDayPlanByID(id int64) (*DayPlan, error) {
    row := db.QueryRow(`SELECT id, user_id, plan_date, status, has_conflicts, conflict_summary,
        generated_at, last_regenerated_at, regenerate_count, feedback_history, prompt_version,
        briefing_id, read_at, created_at, updated_at
        FROM day_plans WHERE id=?`, id)
    return scanDayPlan(row)
}

func (db *DB) ListDayPlans(userID string, limit int) ([]DayPlan, error) {
    if limit <= 0 { limit = 10 }
    rows, err := db.Query(`SELECT id, user_id, plan_date, status, has_conflicts, conflict_summary,
        generated_at, last_regenerated_at, regenerate_count, feedback_history, prompt_version,
        briefing_id, read_at, created_at, updated_at
        FROM day_plans WHERE user_id=? ORDER BY plan_date DESC LIMIT ?`, userID, limit)
    if err != nil { return nil, fmt.Errorf("list day_plans: %w", err) }
    defer rows.Close()
    var out []DayPlan
    for rows.Next() {
        p, err := scanDayPlanRows(rows)
        if err != nil { return nil, err }
        out = append(out, *p)
    }
    return out, rows.Err()
}

func (db *DB) CreateDayPlanItems(planID int64, items []DayPlanItem) error {
    if len(items) == 0 { return nil }
    for _, it := range items {
        if it.Tags == "" { it.Tags = "[]" }
        _, err := db.Exec(`INSERT INTO day_plan_items
            (day_plan_id, kind, source_type, source_id, title, description, rationale,
             start_time, end_time, duration_min, priority, status, order_index, tags)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
            planID, it.Kind, it.SourceType, it.SourceID, it.Title, it.Description, it.Rationale,
            nullTimeStr(it.StartTime), nullTimeStr(it.EndTime), it.DurationMin, it.Priority,
            it.Status, it.OrderIndex, it.Tags)
        if err != nil { return fmt.Errorf("insert day_plan_item: %w", err) }
    }
    return nil
}

// ReplaceAIItems deletes items where source_type is neither 'manual' nor 'calendar'
// and inserts the new items atomically. Manual and calendar items are preserved.
func (db *DB) ReplaceAIItems(planID int64, newItems []DayPlanItem) error {
    tx, err := db.Begin()
    if err != nil { return fmt.Errorf("begin tx: %w", err) }
    defer tx.Rollback()

    if _, err := tx.Exec(`DELETE FROM day_plan_items
        WHERE day_plan_id=? AND source_type NOT IN ('manual','calendar')`, planID); err != nil {
        return fmt.Errorf("delete AI items: %w", err)
    }
    for _, it := range newItems {
        if it.Tags == "" { it.Tags = "[]" }
        _, err := tx.Exec(`INSERT INTO day_plan_items
            (day_plan_id, kind, source_type, source_id, title, description, rationale,
             start_time, end_time, duration_min, priority, status, order_index, tags)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
            planID, it.Kind, it.SourceType, it.SourceID, it.Title, it.Description, it.Rationale,
            nullTimeStr(it.StartTime), nullTimeStr(it.EndTime), it.DurationMin, it.Priority,
            it.Status, it.OrderIndex, it.Tags)
        if err != nil { return fmt.Errorf("insert new item: %w", err) }
    }
    return tx.Commit()
}

func (db *DB) GetDayPlanItems(planID int64) ([]DayPlanItem, error) {
    rows, err := db.Query(`SELECT id, day_plan_id, kind, source_type, source_id, title, description,
        rationale, start_time, end_time, duration_min, priority, status, order_index, tags,
        created_at, updated_at
        FROM day_plan_items WHERE day_plan_id=?
        ORDER BY
            CASE kind WHEN 'timeblock' THEN 0 ELSE 1 END,
            start_time ASC NULLS LAST,
            order_index ASC`, planID)
    if err != nil { return nil, fmt.Errorf("query items: %w", err) }
    defer rows.Close()
    var out []DayPlanItem
    for rows.Next() {
        it, err := scanDayPlanItemRows(rows)
        if err != nil { return nil, err }
        out = append(out, *it)
    }
    return out, rows.Err()
}

func (db *DB) IncrementRegenerateCount(planID int64, feedback string) error {

func scanDayPlan(row *sql.Row) (*DayPlan, error) {
    var p DayPlan

    var hc int

        &genAt, &p.LastRegeneratedAt, &p.RegenerateCount, &p.FeedbackHistory, &p.PromptVersion,

    p.UpdatedAt, _ = time.Parse(time.RFC3339, strings.ReplaceAll(updatedAt, " ", "T"))

        &it.Title, &it.Description, &it.Rationale, &startTime, &endTime, &it.DurationMin,

// gatherTasks returns active user tasks.
func (p *Pipeline) gatherTasks() ([]db.Task, error) {
    return p.db.GetTasks(db.TaskFilter{
        Statuses: []string{"todo", "in_progress", "blocked"},
    })
}

// gatherCalendarEvents returns events whose start_time falls on plan_date (local).
func (p *Pipeline) gatherCalendarEvents(date string) ([]db.CalendarEvent, error) {
    day, err := time.ParseInLocation("2006-01-02", date, time.Local)
    if err != nil { return nil, fmt.Errorf("parse date: %w", err) }
    start := day
    end := day.Add(24 * time.Hour)
    return p.db.GetCalendarEventsInRange(start, end)
}

// gatherBriefing returns today's briefing, or yesterday's as fallback.
func (p *Pipeline) gatherBriefing(userID, date string) *db.Briefing {
    b, _ := p.db.GetBriefing(userID, date)
    if b != nil { return b }
    y, _ := time.Parse("2006-01-02", date)
    yesterday := y.AddDate(0, 0, -1).Format("2006-01-02")
    b, _ = p.db.GetBriefing(userID, yesterday)
    return b
}

// gatherJira returns active Jira issues assigned to the current user.
// Gracefully returns empty slice if the table is absent or query fails.
func (p *Pipeline) gatherJira(userID string) []db.JiraIssue {
    out, err := p.db.GetActiveJiraIssuesForUser(userID)
    if err != nil {
        p.logger.Printf("dayplan: jira unavailable: %v", err)
        return nil
    }
    return out
}

// gatherPeople returns active people cards with red flags.
func (p *Pipeline) gatherPeople() []db.PeopleCard {
    out, err := p.db.GetPeopleCardsWithRedFlags()
    if err != nil {
        p.logger.Printf("dayplan: people cards unavailable: %v", err)
        return nil
    }
    return out
}

// gatherManualItems returns existing items where source_type='manual' — preserved across regeneration.
func (p *Pipeline) gatherManualItems(planID int64) ([]db.DayPlanItem, error) {
    items, err := p.db.GetDayPlanItems(planID)
    if err != nil { return nil, err }
    out := items[:0]
    for _, it := range items {
        if it.SourceType == db.DayPlanItemSourceManual {
            out = append(out, it)
        }
    }
    return out, nil
}

// gatherPreviousPlan returns yesterday's plan (for summary/context).
func (p *Pipeline) gatherPreviousPlan(userID, date string) *db.DayPlan {
    y, _ := time.Parse("2006-01-02", date)
    prev, _ := p.db.GetDayPlan(userID, y.AddDate(0, 0, -1).Format("2006-01-02"))
    return prev
}

// formatBriefingContext extracts attention + coaching from a briefing into the prompt section string.
func formatBriefingContext(b *db.Briefing) string {
    if b == nil { return "(none)" }
    var att []briefing.AttentionItem
    _ = json.Unmarshal([]byte(b.Attention), &att)
    var sb strings.Builder
    if len(att) > 0 {
        sb.WriteString("Attention items:\n")
        for _, a := range att {
            sb.WriteString(fmt.Sprintf("- [%s:%s] %s (priority: %s; reason: %s)\n",
                a.SourceType, a.SourceID, a.Text, a.Priority, a.Reason))
        }
    }
    var coaching []briefing.CoachingItem
    _ = json.Unmarshal([]byte(b.Coaching), &coaching)
    if len(coaching) > 0 {
        sb.WriteString("Coaching hints:\n")
        for _, c := range coaching { sb.WriteString("- "+c.Text+"\n") }
    }
    if sb.Len() == 0 { return "(none)" }
    return sb.String()
}

func formatCalendarSection(events []db.CalendarEvent) string {
    if len(events) == 0 { return "(none)" }
    var sb strings.Builder
    for _, e := range events {
        sb.WriteString(fmt.Sprintf("- [%s–%s] %s\n",

func formatTasksSection(tasks []db.Task) string {
    if len(tasks) == 0 { return "(none)" }
    now := time.Now()
    var sb strings.Builder
    for _, t := range tasks {
        overdue := ""
        if t.DueDate != "" {
            if due, err := time.Parse("2006-01-02", t.DueDate); err == nil && due.Before(now) {
                overdue = " OVERDUE"
            }
        }
        sb.WriteString(fmt.Sprintf("- [id=%d priority=%s due=%s%s status=%s ownership=%s] %s",
            t.ID, t.Priority, t.DueDate, overdue, t.Status, t.Ownership, t.Text))
        if t.Intent != "" { sb.WriteString(fmt.Sprintf("\n  Intent: %s", t.Intent)) }
        if t.Blocking != "" { sb.WriteString(fmt.Sprintf("\n  Blocking: %s", t.Blocking)) }
        sb.WriteString("\n")

    return sb.String()
}

// buildItems validates AI items and converts them into DayPlanItem records.
// Invalid items are dropped; reasons are returned in the `dropped` slice.
func buildItems(r *GenerateResult, date string, events []db.CalendarEvent,
    taskIDs, jiraKeys map[string]bool) ([]db.DayPlanItem, []string) {
    var out []db.DayPlanItem
    var dropped []string

    order := 0
    // Timeblocks
    for _, ai := range r.Timeblocks {
        it, reason := aiToTimeblock(ai, date, events, taskIDs, jiraKeys)
        if reason != "" { dropped = append(dropped, reason); continue }
        it.OrderIndex = order; order++
        out = append(out, *it)
    }
    order = 0
    for _, ai := range r.Backlog {
        it, reason := aiToBacklog(ai, taskIDs, jiraKeys)
        if reason != "" { dropped = append(dropped, reason); continue }
        it.OrderIndex = order; order++
        out = append(out, *it)
    }
    return out, dropped
}

func aiToTimeblock(ai AIItem, date string, events []db.CalendarEvent,
    taskIDs, jiraKeys map[string]bool) (*db.DayPlanItem, string) {
    if ai.SourceType == "calendar" { return nil, "calendar items must not come from AI" }
    if ai.StartTimeLocal == "" || ai.EndTimeLocal == "" {
        return nil, fmt.Sprintf("timeblock %q missing start/end time", ai.Title)
    }
    if reason := validateSource(ai, taskIDs, jiraKeys); reason != "" { return nil, reason }

    start, err := time.ParseInLocation("2006-01-02 15:04", date+" "+ai.StartTimeLocal, time.Local)
    if err != nil { return nil, fmt.Sprintf("parse start_time: %v", err) }
    end, err := time.ParseInLocation("2006-01-02 15:04", date+" "+ai.EndTimeLocal, time.Local)
    if err != nil { return nil, fmt.Sprintf("parse end_time: %v", err) }
    if !end.After(start) { return nil, fmt.Sprintf("timeblock %q end before start", ai.Title) }

    for _, ev := range events {
        if timesOverlap(start, end, ev.StartTime, ev.EndTime) {
            return nil, fmt.Sprintf("timeblock %q overlaps calendar event %q", ai.Title, ev.Title)
        }
    }

    return &db.DayPlanItem{
        Kind:        db.DayPlanItemKindTimeblock,
        SourceType:  ai.SourceType,
        SourceID:    sourceIDToNullString(ai.SourceID),
        Title:       ai.Title,
        Description: strToNull(ai.Description),
        Rationale:   strToNull(ai.Rationale),
        StartTime:   sql.NullTime{Time: start, Valid: true},
        EndTime:     sql.NullTime{Time: end, Valid: true},
        DurationMin: sql.NullInt64{Int64: int64(end.Sub(start).Minutes()), Valid: true},
        Priority:    strToNull(ai.Priority),
        Status:      db.DayPlanItemStatusPending,
        Tags:        "[]",
    }, ""
}

func aiToBacklog(ai AIItem, taskIDs, jiraKeys map[string]bool) (*db.DayPlanItem, string) {
    if ai.SourceType == "calendar" { return nil, "calendar items must not come from AI" }
    if reason := validateSource(ai, taskIDs, jiraKeys); reason != "" { return nil, reason }
    return &db.DayPlanItem{
        Kind:        db.DayPlanItemKindBacklog,
        SourceType:  ai.SourceType,
        SourceID:    sourceIDToNullString(ai.SourceID),
        Title:       ai.Title,
        Description: strToNull(ai.Description),
        Rationale:   strToNull(ai.Rationale),
        Priority:    strToNull(ai.Priority),
        Status:      db.DayPlanItemStatusPending,
        Tags:        "[]",
    }, ""
}

func validateSource(ai AIItem, taskIDs, jiraKeys map[string]bool) string {
    switch ai.SourceType {
    case "task":
        id := sourceIDToString(ai.SourceID)
        if !taskIDs[id] { return fmt.Sprintf("unknown task source_id %q for %q", id, ai.Title) }
    case "jira":
        id := sourceIDToString(ai.SourceID)
        if !jiraKeys[id] { return fmt.Sprintf("unknown jira source_id %q for %q", id, ai.Title) }
    case "focus":
        if ai.SourceID != nil && fmt.Sprintf("%v", ai.SourceID) != "" {
            // tolerate, but clear it
        }
    case "briefing_attention":
        // Accept any non-empty string
        if sourceIDToString(ai.SourceID) == "" { return fmt.Sprintf("briefing_attention item %q missing source_id", ai.Title) }
    default:
        return fmt.Sprintf("invalid source_type %q for %q", ai.SourceType, ai.Title)
    }
    return ""
}

func sourceIDToString(v any) string {
    if v == nil { return "" }
    switch x := v.(type) {
    case string: return x
    case float64: return fmt.Sprintf("%d", int64(x))
    case int: return fmt.Sprintf("%d", x)
    case int64: return fmt.Sprintf("%d", x)
    default: return fmt.Sprintf("%v", x)
    }
}

func sourceIDToNullString(v any) sql.NullString {
    s := sourceIDToString(v)
    if s == "" { return sql.NullString{} }

func (p *Pipeline) syncCalendarItems(planID int64, date string, events []db.CalendarEvent) error {
    existing, err := p.db.GetDayPlanItems(planID)
    if err != nil { return err }

    existingByID := map[string]db.DayPlanItem{}
    for _, it := range existing {
        if it.SourceType == db.DayPlanItemSourceCalendar && it.SourceID.Valid {
            existingByID[it.SourceID.String] = it
        }
    }
    eventByID := map[string]db.CalendarEvent{}
    for _, e := range events { eventByID[e.ID] = e }

    // Delete orphans
    for id, it := range existingByID {
        if _, ok := eventByID[id]; !ok {
            if err := p.db.DeleteDayPlanItem(it.ID); err != nil {
                return fmt.Errorf("delete orphan calendar item: %w", err)
            }
        }
    }

    // Upsert each current event
    for id, ev := range eventByID {
        if cur, ok := existingByID[id]; ok {
            // Update in-place
            _, err := p.db.Exec(`UPDATE day_plan_items SET title=?, description=?, start_time=?, end_time=?,
                duration_min=?, updated_at=strftime('%Y-%m-%dT%H:%M:%SZ','now') WHERE id=?`,
                ev.Title, ev.Description, ev.StartTime.UTC().Format("2006-01-02T15:04:05Z"),
                ev.EndTime.UTC().Format("2006-01-02T15:04:05Z"),
                int64(ev.EndTime.Sub(ev.StartTime).Minutes()), cur.ID)
            if err != nil { return fmt.Errorf("update calendar item: %w", err) }
        } else {
            // Insert new
            it := db.DayPlanItem{
                DayPlanID: planID, Kind: db.DayPlanItemKindTimeblock,
                SourceType: db.DayPlanItemSourceCalendar,
                SourceID:   sql.NullString{String: ev.ID, Valid: true},
                Title:      ev.Title,
                Description: sql.NullString{String: ev.Description, Valid: ev.Description != ""},
                StartTime:  sql.NullTime{Time: ev.StartTime, Valid: true},
                EndTime:    sql.NullTime{Time: ev.EndTime, Valid: true},
                DurationMin: sql.NullInt64{Int64: int64(ev.EndTime.Sub(ev.StartTime).Minutes()), Valid: true},
                Status:     db.DayPlanItemStatusPending,
                Tags:       "[]",
            }
            if err := p.db.CreateDayPlanItems(planID, []db.DayPlanItem{it}); err != nil { return err }
        }
    }

package dayplan

import (
    "context"
    "fmt"
    "strings"
    "time"

// DetectConflicts compares plan timeblocks against current calendar events for date
// and updates has_conflicts + conflict_summary on the plan row.
func (p *Pipeline) DetectConflicts(ctx context.Context, userID, date string) error {
    plan, err := p.db.GetDayPlan(userID, date)
    if err != nil { return err }
    if plan == nil { return nil }

    items, err := p.db.GetDayPlanItems(plan.ID)
    if err != nil { return err }
    events, err := p.gatherCalendarEvents(date)
    if err != nil { return err }

    var conflicts []string
    for _, it := range items {
        if it.Kind != "timeblock" || it.SourceType == "calendar" { continue }
        if !it.StartTime.Valid || !it.EndTime.Valid { continue }
        for _, ev := range events {
            if timesOverlap(it.StartTime.Time, it.EndTime.Time, ev.StartTime, ev.EndTime) {
                conflicts = append(conflicts,
                    fmt.Sprintf("%s (%s–%s) overlaps %q (%s–%s)",
                        it.Title,
                        it.StartTime.Time.Local().Format("15:04"),
                        it.EndTime.Time.Local().Format("15:04"),
                        ev.Title,
                        ev.StartTime.Local().Format("15:04"),
                        ev.EndTime.Local().Format("15:04")))
            }
        }
    }

    summary := ""
    if len(conflicts) > 0 { summary = strings.Join(conflicts, "; ") }
    _ = time.Now()
    return p.db.SetHasConflicts(plan.ID, len(conflicts) > 0, summary)
}

struct ProcessCLIRunner: CLIRunnerProtocol {
    let binaryPath: String  // resolved via Constants.findWatchtowerPath() or similar
    func run(args: [String]) async throws -> Data {
        let p = Process()
        p.executableURL = URL(fileURLWithPath: binaryPath)
        p.arguments = args
        let out = Pipe()
        p.standardOutput = out
        try p.run()
        p.waitUntilExit()
        return out.fileHandleForReading.readDataToEndOfFile()
    }
}
